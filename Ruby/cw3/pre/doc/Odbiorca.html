def ilosc_liczb_parzystych()
    am_of_even = @tab.map {|path| File.read(path)}.map{ |text| text.split() }.map{ |arr| arr.select {|num| (num.to_i % 2).zero?  }}.map {|arr| arr.length() }.sum()
     am_of_even
end
